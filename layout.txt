    <script type="text/javascript">
        var navBar = null, headerBar=null, breadcrumb=[];
        navBar  = new Vue({
            el: "#nav-scroll",
            data: {
                isCollapse: false, //收起
                defIndex: '@(ViewBag.menuIndex==null?"1": ViewBag.menuIndex)',
                mid:@Model.Minisns.Id,
                sPath: [],
                isCreated: false,
                menus: [
                    { title: "首页", url: "", icon: "#icon-dashboard", children: [] },
                    {
                        title: "用户管理", url: "", icon: "#icon-user", children: [
                            { title: "用户管理", url: "/friend/userlist", icon: "", show: true },
                            { title: "用户管理2", url: "/xq/list", icon: "", show: true},
                            ]
                    },                    
                ],
            },
            methods: {
                menuClose(key, arr) {
                    //console.log(key,arr)
                },
                menuOpen(key, arr) {
                    //console.log(key, arr)
                },
                //导航选择事件
                menuSelect(key, paths) {
                    //console.log(key, paths);
                    this.defIndex = key;
                    this.sPath = paths;
                    if (key) {
                        var routers = key.split("-");
                        if (routers) {
                            headerBar && (headerBar.breadcrumb = []);
                            breadcrumb = [];
                            for (var i in routers) {
                                var item = this.menus[routers[0] - 1];
                                var title = i == 0 ? item.title : item.children[routers[i] - 1].title;
                                var url = i == 0 ? item.url : item.children[routers[i] - 1].url;
                                if (headerBar) {
                                    headerBar.breadcrumb.push({ title: title, url: url });
                                } else {
                                    breadcrumb.push({ title: title, url: url });
                                }
                                if (url && paths) {
                                   window.location.href = url + "/" + this.mid;
                                }
                            }
                        }
                    }
                },
            },
            mounted() {
                this.menuSelect(this.defIndex, "");
                this.isCreated = true;
            }
        });
        headerBar = new Vue({
            el: "#header",
            data: {
                breadcrumb: [],
                isCreated: false
            },
            computed: {
                hideMenu() {
                    return navBar.isCollapse;
                }
            },
            methods: {
                toggleMenu() {
                    navBar.isCollapse = !navBar.isCollapse;
                    if (navBar.isCollapse)
                        $("#app .app-container").addClass("hideSidebar");
                    else
                        $("#app .app-container").removeClass("hideSidebar");
                }
            },
            mounted() {
                if (breadcrumb.length > 0) {
                    this.breadcrumb = breadcrumb;
                }
                this.isCreated = true;
            }
        });
    </script>
